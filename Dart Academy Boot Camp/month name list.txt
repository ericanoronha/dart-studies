const List<String> months = [
  "January",
  "February",
  "March",
  "April",
  "May",
  "June",
  "July",
  "August",
  "September",
  "October",
  "November",
  "December"
];

void main() {
  print("Month 6: ${monthName(6)}");
}

String monthName(int monthNumber) {
  assert(monthNumber >= 1 && monthNumber <= 12, "Out of range.");
  return months[monthNumber -1];
}
